public class Bank {
    
    private double amount;
    private double[]depositWithdraw;
    private int count=1;
    
    public Bank(){
        amount=0;
        depositWithdraw[0]=amount;
    }
    
    public Bank(double a){
        amount=a;
        depositWithdraw[0]=amount;
    }
    
    public void deposit(double a){
        amount+=a;
        depositWithdraw[count]=amount;
        count++;
    }
    
    public void withdraw(double a){
        amount-=a;
        depositWithdraw[count]=(-1)*amount;
        count++;
    }
    
    public void listInteractions(){
        for(int i=0;i<=count;i++){
            System.out.println(depositWithdraw[i]);
        }
    }
    
}
